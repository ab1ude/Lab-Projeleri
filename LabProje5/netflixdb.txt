CREATE DATABASE IF NOT EXISTS netflixdb;
USE netflixdb;

CREATE TABLE IF NOT EXISTS Rol (
    id INT AUTO_INCREMENT PRIMARY KEY,
    ad VARCHAR(50) NOT NULL UNIQUE
);

CREATE TABLE IF NOT EXISTS Kullanici (
    id INT AUTO_INCREMENT PRIMARY KEY,
    ad VARCHAR(50) NOT NULL,
    soyad VARCHAR(50) NOT NULL,
    email VARCHAR(100) NOT NULL UNIQUE,
    sifre VARCHAR(255) NOT NULL,
    dogum_tarihi DATE NOT NULL,
    cinsiyet VARCHAR(10),
    ulke VARCHAR(50),
    rol_id INT,
    aktif_mi BOOLEAN DEFAULT TRUE,
    FOREIGN KEY (rol_id) REFERENCES Rol(id)
);

CREATE TABLE IF NOT EXISTS Program (
    id INT AUTO_INCREMENT PRIMARY KEY,
    ad VARCHAR(255) NOT NULL,
    tip VARCHAR(10) NOT NULL,
    aciklama TEXT,
    yayin_yili INT,
    uzunluk INT,
    bolum_sayisi INT DEFAULT 1,
    ortalama_puan FLOAT DEFAULT 0.0,
    izlenme_sayisi INT DEFAULT 0
);

CREATE TABLE IF NOT EXISTS Tur (
    id INT AUTO_INCREMENT PRIMARY KEY,
    ad VARCHAR(50) NOT NULL UNIQUE
);

CREATE TABLE IF NOT EXISTS ProgramTur (
    program_id INT,
    tur_id INT,
    PRIMARY KEY (program_id, tur_id),
    FOREIGN KEY (program_id) REFERENCES Program(id) ON DELETE CASCADE,
    FOREIGN KEY (tur_id) REFERENCES Tur(id) ON DELETE CASCADE
);

CREATE TABLE IF NOT EXISTS KullaniciProgram (
    kullanici_id INT,
    program_id INT,
    verilen_puan INT CHECK (verilen_puan BETWEEN 1 AND 10),
    PRIMARY KEY (kullanici_id, program_id),
    FOREIGN KEY (kullanici_id) REFERENCES Kullanici(id) ON DELETE CASCADE,
    FOREIGN KEY (program_id) REFERENCES Program(id) ON DELETE CASCADE
);

CREATE TABLE IF NOT EXISTS Favori (
    id INT AUTO_INCREMENT PRIMARY KEY,
    kullanici_id INT,
    program_id INT,
    FOREIGN KEY (kullanici_id) REFERENCES Kullanici(id) ON DELETE CASCADE,
    FOREIGN KEY (program_id) REFERENCES Program(id) ON DELETE CASCADE
);

CREATE TABLE IF NOT EXISTS KullaniciTur (
    kullanici_id INT,
    tur_id INT,
    PRIMARY KEY (kullanici_id, tur_id),
    FOREIGN KEY (kullanici_id) REFERENCES Kullanici(id) ON DELETE CASCADE,
    FOREIGN KEY (tur_id) REFERENCES Tur(id) ON DELETE CASCADE
);

CREATE TABLE IF NOT EXISTS OturumLog (
    id INT AUTO_INCREMENT PRIMARY KEY,
    kullanici_id INT,
    giris_tarihi DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (kullanici_id) REFERENCES Kullanici(id) ON DELETE CASCADE
);

CREATE TABLE IF NOT EXISTS IzlemeLog (
    id INT AUTO_INCREMENT PRIMARY KEY,
    kullanici_id INT,
    program_id INT,
    bolum_no INT DEFAULT 1,
    izleme_tarihi DATETIME DEFAULT CURRENT_TIMESTAMP,
    izlenen_sure INT DEFAULT 0,
    tamamlandi_mi BOOLEAN DEFAULT FALSE,
    FOREIGN KEY (kullanici_id) REFERENCES Kullanici(id) ON DELETE CASCADE,
    FOREIGN KEY (program_id) REFERENCES Program(id) ON DELETE CASCADE
);
